df = pd.read_excel(fpath, sheet_name='Sheet1')
df[''] = pd.to_datetime(df[''], errors='coerce')
def classify_season(month):
    if month in [12, 1, 2]:
        return ''
    elif month in [3, 4, 5]:
        return ''
    elif month in [6, 7, 8]:
        return ''
    else:
        return ''

#
df[''] = df[''].dt.month.apply(classify_season)
# 
filtered_df = df[df[''] == '']
# 
seasonal_avg_prices = filtered_df.groupby('')[''].mean().reset_index()

# 
plt.figure(figsize=(10, 6))
for i, row in seasonal_avg_prices.iterrows():
    plt.scatter(row[''], row[''], s=100, label=row[''])

plt.title('')
plt.xlabel('')
plt.ylabel('')
plt.grid(True)
plt.legend(title='')
plt.tight_layout()
plt.show()
